CREATE TABLE `members` (
	`memid` VARCHAR(9) NOT NULL,
	`memname` VARCHAR(20) NOT NULL,
	`passwd` VARCHAR(128) NULL DEFAULT NULL,
	`passwdmdt` DATETIME NULL DEFAULT NULL,
	`jumin` VARCHAR(64) NULL DEFAULT NULL,
	`addr` VARCHAR(100) NULL DEFAULT NULL,
	`birthday` DATE NULL DEFAULT NULL,
	`jobcd` CHAR(1) NULL DEFAULT NULL,
	`mileage` DECIMAL(7,0) UNSIGNED NULL DEFAULT '0',
	`stat` ENUM('Y','N') NOT NULL DEFAULT 'Y',
	`enterdtm` DATETIME NOT NULL DEFAULT current_timestamp(),
	`leavedtm` DATETIME NULL DEFAULT NULL,
	PRIMARY KEY (`memid`)
);
